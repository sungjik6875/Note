##### action.js

```javascript
// 기간 계산하기
const calculate = function() {
  // 입력값을 Date 객체로 변환
  const firstDayInput = document.querySelector("#input>input").value;
  let year = firstDayInput.slice(0, 4);
  let month = firstDayInput.slice(4, 6);
  let day = firstDayInput.slice(6, 8);
  const firstDay = new Date(`${year}-${month}-${day}`);
  
  // passedDay 계산하기 (passedTime / (밀리초*초*분*시))
  const now = new Date();
  const passedTime = now - firstDay
  const passedDay = Math.round(passedTime/(1000*60*60*24));

  // 결과 보여주기
  const passedDayResult = document.querySelector("#passedDay");
  passedDayResult.innerText = `우리 만난지 ${passedDay}일`;
}

// 위의 함수 호출
const showResult = function() {
  calculate();
}
```
