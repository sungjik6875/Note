##### action.js

```javascript

const list = [];

const form = document.getElementById("form");
const itemList = document.getElementById("itemList");

const showList = function(mode) {
  if (mode === 'init') {
    for (let i=0; i < list.length; i++) {
      itemList.innerHTML += `<li>${list[i]}</li>`
    }
  } else {
    itemList.innerHTML += `<li>${list[list.length-1]}</li>`
  }
};

const addItem = function(e) {
  e.preventDefault();
  const itemInput = document.querySelector("#form>input");
  let item = itemInput.value;

  if (item !== "") {
    list.push(item);
    showList('add');
  } else {
    alert("아이템을 입력하세요.")
  }
  itemInput.value = "";
  itemInput.focus();
};

showList('init');
form.addEventListener("submit", addItem);
```
