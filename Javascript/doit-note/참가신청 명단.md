##### action.js

```javascript
const newRegister = function(e) {
  e.preventDefault();
  const newNameList = document.createElement("li");
  const newName = document.querySelector("#userName").value;
  newNameList.innerHTML = `${newName}<span>[X]</span>`;
  
  let userList = document.querySelector("#nameList");
  userList.appendChild(newNameList);

  const delBtn = newNameList.childNodes[1]
  delBtn.addEventListener('click', deleteThis);
}

const deleteThis = function() {
  this.parentNode.parentNode.removeChild(this.parentNode);
}

const regBtn = document.querySelector("#myBtn");
regBtn.addEventListener('click', newRegister);
```
