##### action.js

```javascript
// 전공 선택 안내메시지 보여주기
const majorSelectOption = document.forms[0].elements.major
majorSelectOption.addEventListener("change", function() {
  const major = majorSelectOption.options[majorSelectOption.selectedIndex].innerText;
  alert(`[${major}] 전공을 선택하셨습니다.`)
})
```
