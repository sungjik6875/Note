##### index.html

```html
<body>
  <span>원가</span>
  <input id="priceInput" type="number">
  <span>할인율</span>
  <input id="ratioInput" type="number"><span>%</span>

  <button id="myBtn">할인가격 계산하기</button>
  <div id="result"></div>
  
  <script src="./js/action.js"></script>
  
</body>
```



##### action.js

```javascript
const calc = function() {
  const price = parseInt(document.getElementById("priceInput").value)
  const ratio = parseInt(document.getElementById("ratioInput").value)
  const priceForSale = (price * (1 - ratio/100))

  const result = document.getElementById("result")
  result.innerHTML = `원가는 ${price}, 할인율은 ${ratio}, 할인된 가격은 <span style="color: red">${priceForSale}</span>`
}

var myBtn = document.querySelector('#myBtn')
myBtn.onclick = function() {
  calc()
}
```
