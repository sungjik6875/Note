##### action.js

```javascript
// 아이디 유효성 검증 - 자릿수
const validateId = function() {
  const id = document.forms[0].elements.id;
  if (id.value.length >= 4 && id.value.length <=15) {
    return true
  } else {
    return false
  }
}

// 비밀번호 유효성 검증 - 자릿수
const validatePw1 = function() {
  const password = document.forms[0].elements.password;
  if (password.value.length >= 8 && password.value.length <= 20) {
    return true
  } else {
    return false
  }
}

// 비밀번호 유효성 검증 - 두 값의 일치
const validatePw2 = function() {
  const password = document.forms[0].elements.password;
  const passwordConfirm = document.forms[0].elements.passwordConfirm;
  if (password.value === passwordConfirm.value) {
    return true
  } else {
    return false
  }
}

// 입력값 유효성 검증
const validate = function(e) {
  e.preventDefault();
  if (validateId() === false) {
    alert("아이디는 4~15자리 이어야 합니다.")
  } else if (validatePw1() === false) {
    alert("비밀번호는 8~20자리 이어야 합니다.")
  } else if (validatePw2() === false) {
    alert("비밀번호가 일치하지 않습니다.")
  } else {
    alert("회원가입이 완료되었습니다.")
  }
}

const signUpBtn = document.getElementById("signUpBtn");
signUpBtn.addEventListener("click", validate);
```
