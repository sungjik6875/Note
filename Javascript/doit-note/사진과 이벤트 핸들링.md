##### action.js

```javascript
const bugImg = document.querySelector("img");
bugImg.onmouseover = function() {
  bugImg.style.border = "1px solid black"
}
bugImg.onmouseleave = function() {
  bugImg.style.border = ""
}


const desc = document.getElementById("desc");
const showBtn = document.getElementById("showBtn");
showBtn.onclick = function() { 
  if (desc.style.visibility === "visible") {
    desc.style.visibility = "hidden";
    showBtn.innerText = "설명 보기";
  } else {
    desc.style.visibility = "visible";
    showBtn.innerText = "설명 숨기기";
  }
}
```
